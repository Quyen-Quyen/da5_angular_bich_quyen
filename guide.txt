--- Sử dụng git----

-- thao tác up code---
git status 
--kiểm tra sự thay đổi của code

git add .
--- lưu toàn bộ code vào nhanh git(local)

git commit -m"ghi thông tin cần node"
--- lưu thông tin node lên git

git push origin bichquyen 
--- đẩy code lên nhánh bichquyen ( đẩy lên github)

 git checkout main

  ---quay lại nhánh của mình

b4 : git merge main
